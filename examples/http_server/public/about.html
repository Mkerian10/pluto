How it works

This server is written entirely in Pluto. It uses three standard library modules:

  • std.net — TCP listener and connections
  • std.fs — Reading files from disk
  • std.strings — Parsing HTTP requests

The server code

import std.net
import std.fs
import std.strings

fn main() {
    let server = net.listen("127.0.0.1", 8080)
    print("Listening on http://127.0.0.1:8080")

    while true {
        let conn = server.accept()
        let request = conn.read(4096)
        let path = parse_request_path(request)
        serve_file(conn, "public", path)
    }
}
